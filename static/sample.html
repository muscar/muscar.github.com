#include <iostream>
#include <string>

class Person
{
    std::string name_;
    int age_;

public:
    Person(std::string name, int age)
        : name_(name), age_(age)
    {
    }

    void show()
    {
        std::cout << "Name: " << name_ << "; "
                  << "Age: " << age_ << "; "
                  << "Description: " << describe() << std::endl;
    }

protected:
    std::string describe() const
    {
        return "person";
    }
};

class Student : public Person
{
    std::string faculty_;

public:
    Student(std::string name, int age, std::string faculty)
        : Person(name, age), faculty_(faculty)
    {
    }

protected:
    std::string describe() const
    {
        return "student";
    }
};

int main(int argc, char const *argv[])
{
    Person p("Andrei", 20);
    Student s("Anda", 19, "ACE");

    p.show();
    s.show();

    return 0;
}
